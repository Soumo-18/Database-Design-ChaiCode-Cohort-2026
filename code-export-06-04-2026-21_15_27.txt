title Insta THRIFT Creator Store
customer [icon: user]{
  id SERIAL PK
  name VARCHAR(50) NOT NULL
  phone VARCHAR(12) UNIQUE NOT NULL
  email VARCHAR(322) UNIQUE NOT NULL
  instagram_handle VARCHAR(50) UNIQUE NOT NULL
  address string NOT NULL
}
products [icon: shopping-bag]{
  id SERIAL PK
  name string NOT NULL
  description string
  product_type enum(thrifted, handmade)
  category enum(Tops, Bottoms, Accessories, Home, Jewelry)
  price INT NOT NULL
  size enum(XS, S, M, L, XL, Free_Size)
  colour VARCHAR(30) CHECK(IN array) NULL
  condition enum(new_with_tags, excellent, good, fair)
  image_url TEXT
  video_url TEXT
  is_Active BOOLEAN DEFAULT TRUE
}

inventory[icon:shopping-bag]{
  id SERIAL PK
  product_id INT FK NOT NULL
  quantity_available INT DEFAULT 1
  last_updated timestamp DEFAULT now() 
}

orders[icon: shopping-cart] {
  id SERIAL PK
  customer_id INT FK NOT NULL
  order_date timestamp
  total_amount INT NOT NULL
  rating INT NULL 
}

order_items[icon:dart] {
  id SERIAL PK
  order_id INT FK NOT NULL
  product_id INT FK NOT NULL
  quantity INT NOT NULL DEFAULT 1
  unit_price INT NOT NULL
}

payments[icon: google-pay]{
  id SERIAL PK
  order_id INT FK NOT NULL
  payment_method enum(UPI, Card, COD)
  amount_paid INT NOT NULL
  payment_status enum(Pending, Paid, Failed)
  payment_date timestamp DEFAULT NOW()
}

shipping[icon: car]{
  id SERIAL PK
  order_id INT FK NOT NULL
  shipping_address string NOT NULL
  shipping_status enum(confirmed, out_for_delivery ,delivered)
  
}

// 1 Customer can place Many Orders
customer.id < orders.customer_id


// 1 Order TO Many Order Items
orders.id < order_items.order_id 

// 1 Product to Many Order Items
products.id < order_items.product_id

// 1 Product to 1 Inventory Record
products.id - inventory.product_id

// 1 Order to 1 Payment 
orders.id < payments.order_id //One order can have MULTIPLE rows in the payments table(let consider if the payment fails for the time )

// 1 Order to 1 Shipping Record
orders.id - shipping.order_id
